CheckoutService.cs 
public async Task<OrderResult> CompleteCheckoutAsync(CartDto cart)
{
    using var tx = await _db.Database
        .BeginTransactionAsync();

    var stock = await _inventory
        .ReserveAsync(cart.Items);

    if (!stock.IsAvailable)
        return OrderResult.OutOfStock();

    var order = await _payments
        .ChargeAsync(cart.Total);

    await tx.CommitAsync();
    return OrderResult.Success(order);
}
